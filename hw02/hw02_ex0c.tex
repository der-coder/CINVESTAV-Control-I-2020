\item Find a transformation matrix $S$ such that
\begin{equation*}
 S^{-1} A S = J
\end{equation*}
where
\begin{equation*}
 A = \begin{bmatrix}
      4 & 1 & -2\\
      1 & 0 & 2\\
      1 & -1 & 3
     \end{bmatrix}
, \qquad
J = \begin{bmatrix}
     3 & 1 & 0\\
     0 & 3 & 0\\
     0 & 0 & 1
    \end{bmatrix}
\end{equation*}
